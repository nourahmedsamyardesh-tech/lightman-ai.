async function sendMessage() { const input = document.getElementById('user-input'); const box = document.getElementById('chat-box'); const text = input.value; if(text) { box.innerHTML += `

`; input.value = ''; const response = await fetch('/api/chat', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ message: text }) }); const data = await response.json(); box.innerHTML += `
